 96: PetscErrorCode PetscOpenHistoryFile(const char filename[], FILE **fd)
 97: {
 98:   PetscMPIInt rank, size;
 99:   char        pfile[PETSC_MAX_PATH_LEN], pname[PETSC_MAX_PATH_LEN], fname[PETSC_MAX_PATH_LEN], date[64];
100:   char        version[256];

102:   PetscFunctionBegin;
103:   PetscCallMPI(MPI_Comm_rank(PETSC_COMM_WORLD, &rank));
104:   if (rank == 0) {
105:     char arch[10];

107:     PetscCall(PetscGetArchType(arch, 10));
108:     PetscCall(PetscGetDate(date, 64));
109:     PetscCall(PetscGetVersion(version, 256));
110:     PetscCallMPI(MPI_Comm_size(PETSC_COMM_WORLD, &size));
111:     if (filename) {
112:       PetscCall(PetscFixFilename(filename, fname));
113:     } else {
114:       PetscCall(PetscGetHomeDirectory(pfile, sizeof(pfile)));
115:       PetscCall(PetscStrlcat(pfile, "/.petschistory", sizeof(pfile)));
116:       PetscCall(PetscFixFilename(pfile, fname));
117:     }

119:     *fd = fopen(fname, "a");
120:     PetscCheck(fd, PETSC_COMM_SELF, PETSC_ERR_FILE_OPEN, "Cannot open file: %s", fname);

122:     PetscCall(PetscFPrintf(PETSC_COMM_SELF, *fd, "----------------------------------------\n"));
123:     PetscCall(PetscFPrintf(PETSC_COMM_SELF, *fd, "%s %s\n", version, date));
124:     PetscCall(PetscGetProgramName(pname, sizeof(pname)));
125:     PetscCall(PetscFPrintf(PETSC_COMM_SELF, *fd, "%s on a %s, %d proc. with options:\n", pname, arch, size));
126:     PetscCall(PetscFPrintf(PETSC_COMM_SELF, *fd, "----------------------------------------\n"));

128:     PetscCall(PetscFFlush(*fd));
129:   }
130:   PetscFunctionReturn(PETSC_SUCCESS);
131: }

133: PETSC_INTERN PetscErrorCode PetscCloseHistoryFile(FILE **fd)
134: {
135:   PetscMPIInt rank;

137:   PetscFunctionBegin;
138:   PetscCallMPI(MPI_Comm_rank(PETSC_COMM_WORLD, &rank));
139:   if (rank == 0) {
140:     char date[64];
141:     int  err;

143:     PetscCall(PetscGetDate(date, sizeof(date)));
144:     PetscCall(PetscFPrintf(PETSC_COMM_SELF, *fd, "----------------------------------------\n"));
145:     PetscCall(PetscFPrintf(PETSC_COMM_SELF, *fd, "Finished at %s\n", date));
146:     PetscCall(PetscFPrintf(PETSC_COMM_SELF, *fd, "----------------------------------------\n"));
147:     PetscCall(PetscFFlush(*fd));
148:     err = fclose(*fd);
149:     PetscCheck(!err, PETSC_COMM_SELF, PETSC_ERR_SYS, "fclose() failed on file");
150:   }
151:   PetscFunctionReturn(PETSC_SUCCESS);
152: }

156: /*
157:    This is ugly and probably belongs somewhere else, but I want to
158:   be able to put a true MPI abort error handler with command line args.

160:     This is so MPI errors in the debugger will leave all the stack
161:   frames. The default MP_Abort() cleans up and exits thus providing no useful information
162:   in the debugger hence we call abort() instead of MPI_Abort().
163: */

165: void Petsc_MPI_AbortOnError(PETSC_UNUSED MPI_Comm *comm, PetscMPIInt *flag, ...)
166: {
167:   PetscFunctionBegin;
168:   PetscCallContinue((*PetscErrorPrintf)("MPI error %d\n", *flag));
169:   abort();
170: }

172: void Petsc_MPI_DebuggerOnError(MPI_Comm *comm, PetscMPIInt *flag, ...)
173: {
174:   PetscFunctionBegin;
175:   PetscCallContinue((*PetscErrorPrintf)("MPI error %d\n", *flag));
176:   if (PetscAttachDebugger()) PETSCABORT(*comm, (PetscErrorCode)*flag); /* hopeless so get out */
177: }

238: PETSC_INTERN PetscErrorCode PetscOptionsCheckInitial_Private(const char help[])
239: {
240:   char        string[64];
241:   MPI_Comm    comm = PETSC_COMM_WORLD;
242:   PetscBool   flg1 = PETSC_FALSE, flg2 = PETSC_FALSE, flag, hasHelp;
243:   PetscBool   checkstack = PETSC_FALSE;
244:   PetscReal   si;
245:   PetscInt    intensity;
246:   int         i;
247:   PetscMPIInt rank;
248:   char        version[256];
249: #if defined(PETSC_USE_LOG)
250:   char              mname[PETSC_MAX_PATH_LEN];
251:   PetscViewerFormat format;
252:   PetscBool         flg4 = PETSC_FALSE;
253: #endif

255:   PetscFunctionBegin;
256:   PetscCallMPI(MPI_Comm_rank(comm, &rank));

258:   if (PetscDefined(USE_DEBUG) && !PetscDefined(HAVE_THREADSAFETY)) checkstack = PETSC_TRUE;
259:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-checkstack", &checkstack, NULL));
260:   PetscCall(PetscStackSetCheck(checkstack));

262:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-checkfunctionlist", &PetscPrintFunctionList, NULL));

264: #if !defined(PETSC_HAVE_THREADSAFETY)
265:   if (!(PETSC_RUNNING_ON_VALGRIND)) {
266:     /*
267:       Setup the memory management; support for tracing malloc() usage
268:     */
269:     PetscBool mdebug = PETSC_FALSE, eachcall = PETSC_FALSE, initializenan = PETSC_FALSE, mlog = PETSC_FALSE;
270:     PetscBool flg3 = PETSC_FALSE;

272:     if (PetscDefined(USE_DEBUG)) {
273:       mdebug        = PETSC_TRUE;
274:       initializenan = PETSC_TRUE;
275:       PetscCall(PetscOptionsHasName(NULL, NULL, "-malloc_test", &flg1));
276:     } else {
277:       /* don't warn about unused option */
278:       PetscCall(PetscOptionsHasName(NULL, NULL, "-malloc_test", &flg1));
279:       flg1 = PETSC_FALSE;
280:     }
281:     PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc_debug", &flg2, &flg3));
282:     if (flg1 || flg2) {
283:       mdebug        = PETSC_TRUE;
284:       eachcall      = PETSC_TRUE;
285:       initializenan = PETSC_TRUE;
286:     } else if (flg3 && !flg2) {
287:       mdebug        = PETSC_FALSE;
288:       eachcall      = PETSC_FALSE;
289:       initializenan = PETSC_FALSE;
290:     }

292:     PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc_requested_size", &flg1, &flg2));
293:     if (flg2) PetscCall(PetscMallocLogRequestedSizeSet(flg1));

295:     PetscCall(PetscOptionsHasName(NULL, NULL, "-malloc_view", &mlog));
296:     if (mlog) mdebug = PETSC_TRUE;
297:     /* the next line is deprecated */
298:     PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc", &mdebug, NULL));
299:     PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc_dump", &mdebug, NULL));
300:     PetscCall(PetscOptionsGetBool(NULL, NULL, "-log_view_memory", &mdebug, NULL));
301:     if (mdebug) PetscCall(PetscMallocSetDebug(eachcall, initializenan));
302:     if (mlog) {
303:       PetscReal logthreshold = 0;
304:       PetscCall(PetscOptionsGetReal(NULL, NULL, "-malloc_view_threshold", &logthreshold, NULL));
305:       PetscCall(PetscMallocViewSet(logthreshold));
306:     }
307:   #if defined(PETSC_USE_LOG)
308:     PetscCall(PetscOptionsGetBool(NULL, NULL, "-log_view_memory", &PetscLogMemory, NULL));
309:   #endif
310:   }

312:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc_coalesce", &flg1, &flg2));
313:   if (flg2) PetscCall(PetscMallocSetCoalesce(flg1));
314:   flg1 = PETSC_FALSE;
315:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc_hbw", &flg1, NULL));
316:   /* ignore this option if malloc is already set */
317:   if (flg1 && !petscsetmallocvisited) PetscCall(PetscSetUseHBWMalloc_Private());

319:   flg1 = PETSC_FALSE;
320:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-malloc_info", &flg1, NULL));
321:   if (!flg1) {
322:     flg1 = PETSC_FALSE;
323:     PetscCall(PetscOptionsGetBool(NULL, NULL, "-memory_view", &flg1, NULL));
324:   }
325:   if (flg1) PetscCall(PetscMemorySetGetMaximumUsage());
326: #endif

328: #if defined(PETSC_USE_LOG)
329:   PetscCall(PetscOptionsHasName(NULL, NULL, "-objects_dump", &PetscObjectsLog));
330: #endif

332:   /*
333:       Set the display variable for graphics
334:   */
335:   PetscCall(PetscSetDisplay());

337:   /*
338:      Print main application help message
339:   */
340:   PetscCall(PetscOptionsHasHelp(NULL, &hasHelp));
341:   if (help && hasHelp) {
342:     PetscCall(PetscPrintf(comm, "%s", help));
343:     PetscCall(PetscPrintf(comm, "----------------------------------------\n"));
344:   }

346:   /*
347:       Print the PETSc version information
348:   */
349:   PetscCall(PetscOptionsHasName(NULL, NULL, "-version", &flg1));
350:   if (flg1 || hasHelp) {
351:     /*
352:        Print "higher-level" package version message
353:     */
354:     if (PetscExternalVersionFunction) PetscCall((*PetscExternalVersionFunction)(comm));

356:     PetscCall(PetscGetVersion(version, 256));
357:     PetscCall((*PetscHelpPrintf)(comm, "%s\n", version));
358:     PetscCall((*PetscHelpPrintf)(comm, "%s", PETSC_AUTHOR_INFO));
359:     PetscCall((*PetscHelpPrintf)(comm, "See https://petsc.org/release/changes for recent updates.\n"));
360:     PetscCall((*PetscHelpPrintf)(comm, "See https://petsc.org/release/faq for problems.\n"));
361:     PetscCall((*PetscHelpPrintf)(comm, "See https://petsc.org/release/manualpages for help. \n"));
362:     PetscCall((*PetscHelpPrintf)(comm, "Libraries linked from %s\n", PETSC_LIB_DIR));
363:     PetscCall((*PetscHelpPrintf)(comm, "----------------------------------------\n"));
364:   }

366:   /*
367:        Print "higher-level" package help message
368:   */
369:   if (hasHelp) {
370:     PetscBool hasHelpIntro;

372:     if (PetscExternalHelpFunction) PetscCall((*PetscExternalHelpFunction)(comm));
373:     PetscCall(PetscOptionsHasHelpIntro_Internal(NULL, &hasHelpIntro));
374:     if (hasHelpIntro) {
375:       PetscCall(PetscOptionsDestroyDefault());
376:       PetscCall(PetscFreeMPIResources());
377:       PetscCallMPI(MPI_Finalize());
378:       exit(0);
379:     }
380:   }

382:   /*
383:       Setup the error handling
384:   */
385:   flg1 = PETSC_FALSE;
386:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-on_error_abort", &flg1, NULL));
387:   if (flg1) {
388:     PetscCallMPI(MPI_Comm_set_errhandler(comm, MPI_ERRORS_ARE_FATAL));
389:     PetscCall(PetscPushErrorHandler(PetscAbortErrorHandler, NULL));
390:   }
391:   flg1 = PETSC_FALSE;
392:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-on_error_mpiabort", &flg1, NULL));
393:   if (flg1) PetscCall(PetscPushErrorHandler(PetscMPIAbortErrorHandler, NULL));
394:   flg1 = PETSC_FALSE;
395:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-mpi_return_on_error", &flg1, NULL));
396:   if (flg1) PetscCallMPI(MPI_Comm_set_errhandler(comm, MPI_ERRORS_RETURN));
397:   flg1 = PETSC_FALSE;
398:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-no_signal_handler", &flg1, NULL));
399:   if (!flg1) PetscCall(PetscPushSignalHandler(PetscSignalHandlerDefault, (void *)0));

401:   /*
402:       Setup debugger information
403:   */
404:   PetscCall(PetscSetDefaultDebugger());
405:   PetscCall(PetscOptionsGetString(NULL, NULL, "-on_error_attach_debugger", string, sizeof(string), &flg1));
406:   if (flg1) {
407:     MPI_Errhandler err_handler;

409:     PetscCall(PetscSetDebuggerFromString(string));
410:     PetscCallMPI(MPI_Comm_create_errhandler(Petsc_MPI_DebuggerOnError, &err_handler));
411:     PetscCallMPI(MPI_Comm_set_errhandler(comm, err_handler));
412:     PetscCall(PetscPushErrorHandler(PetscAttachDebuggerErrorHandler, NULL));
413:   }
414:   PetscCall(PetscOptionsGetString(NULL, NULL, "-debug_terminal", string, sizeof(string), &flg1));
415:   if (flg1) PetscCall(PetscSetDebugTerminal(string));
416:   PetscCall(PetscOptionsGetString(NULL, NULL, "-start_in_debugger", string, sizeof(string), &flg1));
417:   PetscCall(PetscOptionsGetString(NULL, NULL, "-stop_for_debugger", string, sizeof(string), &flg2));
418:   if (flg1 || flg2) {
419:     PetscMPIInt    size;
420:     PetscInt       lsize, *ranks;
421:     MPI_Errhandler err_handler;
422:     /*
423:        we have to make sure that all processors have opened
424:        connections to all other processors, otherwise once the
425:        debugger has stated it is likely to receive a SIGUSR1
426:        and kill the program.
427:     */
428:     PetscCallMPI(MPI_Comm_size(comm, &size));
429:     if (size > 2) {
430:       PetscMPIInt dummy = 0;
431:       MPI_Status  status;
432:       for (i = 0; i < size; i++) {
433:         if (rank != i) PetscCallMPI(MPI_Send(&dummy, 1, MPI_INT, i, 109, comm));
434:       }
435:       for (i = 0; i < size; i++) {
436:         if (rank != i) PetscCallMPI(MPI_Recv(&dummy, 1, MPI_INT, i, 109, comm, &status));
437:       }
438:     }
439:     /* check if this processor node should be in debugger */
440:     PetscCall(PetscMalloc1(size, &ranks));
441:     lsize = size;
442:     /* Deprecated in 3.14 */
443:     PetscCall(PetscOptionsGetIntArray(NULL, NULL, "-debugger_nodes", ranks, &lsize, &flag));
444:     if (flag) {
445:       const char *const quietopt = "-options_suppress_deprecated_warnings";
446:       char              msg[4096];
447:       PetscBool         quiet = PETSC_FALSE;

449:       PetscCall(PetscOptionsGetBool(NULL, NULL, quietopt, &quiet, NULL));
450:       if (!quiet) {
451:         PetscCall(PetscStrncpy(msg, "** PETSc DEPRECATION WARNING ** : the option ", sizeof(msg)));
452:         PetscCall(PetscStrlcat(msg, "-debugger_nodes", sizeof(msg)));
453:         PetscCall(PetscStrlcat(msg, " is deprecated as of version ", sizeof(msg)));
454:         PetscCall(PetscStrlcat(msg, "3.14", sizeof(msg)));
455:         PetscCall(PetscStrlcat(msg, " and will be removed in a future release.", sizeof(msg)));
456:         PetscCall(PetscStrlcat(msg, " Please use the option ", sizeof(msg)));
457:         PetscCall(PetscStrlcat(msg, "-debugger_ranks", sizeof(msg)));
458:         PetscCall(PetscStrlcat(msg, " instead.", sizeof(msg)));
459:         PetscCall(PetscStrlcat(msg, " (Silence this warning with ", sizeof(msg)));
460:         PetscCall(PetscStrlcat(msg, quietopt, sizeof(msg)));
461:         PetscCall(PetscStrlcat(msg, ")\n", sizeof(msg)));
462:         PetscCall(PetscPrintf(comm, "%s", msg));
463:       }
464:     } else {
465:       lsize = size;
466:       PetscCall(PetscOptionsGetIntArray(NULL, NULL, "-debugger_ranks", ranks, &lsize, &flag));
467:     }
468:     if (flag) {
469:       for (i = 0; i < lsize; i++) {
470:         if (ranks[i] == rank) {
471:           flag = PETSC_FALSE;
472:           break;
473:         }
474:       }
475:     }
476:     if (!flag) {
477:       PetscCall(PetscSetDebuggerFromString(string));
478:       PetscCall(PetscPushErrorHandler(PetscAbortErrorHandler, NULL));
479:       if (flg1) {
480:         PetscCall(PetscAttachDebugger());
481:       } else {
482:         PetscCall(PetscStopForDebugger());
483:       }
484:       PetscCallMPI(MPI_Comm_create_errhandler(Petsc_MPI_AbortOnError, &err_handler));
485:       PetscCallMPI(MPI_Comm_set_errhandler(comm, err_handler));
486:     } else {
487:       PetscCall(PetscWaitOnError());
488:     }
489:     PetscCall(PetscFree(ranks));
490:   }

492:   PetscCall(PetscOptionsGetString(NULL, NULL, "-on_error_emacs", emacsmachinename, sizeof(emacsmachinename), &flg1));
493:   if (flg1 && rank == 0) PetscCall(PetscPushErrorHandler(PetscEmacsClientErrorHandler, emacsmachinename));

495:     /*
496:         Setup profiling and logging
497:   */
498: #if defined(PETSC_USE_INFO)
499:   {
500:     PetscCall(PetscInfoSetFromOptions(NULL));
501:   }
502: #endif
503:   PetscCall(PetscDetermineInitialFPTrap());
504:   flg1 = PETSC_FALSE;
505:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-fp_trap", &flg1, &flag));
506:   if (flag) PetscCall(PetscSetFPTrap(flg1 ? PETSC_FP_TRAP_ON : PETSC_FP_TRAP_OFF));
507:   PetscCall(PetscOptionsGetInt(NULL, NULL, "-check_pointer_intensity", &intensity, &flag));
508:   if (flag) PetscCall(PetscCheckPointerSetIntensity(intensity));
509: #if defined(PETSC_USE_LOG)
510:   mname[0] = 0;
511:   PetscCall(PetscOptionsGetString(NULL, NULL, "-history", mname, sizeof(mname), &flg1));
512:   if (flg1) {
513:     if (mname[0]) {
514:       PetscCall(PetscOpenHistoryFile(mname, &petsc_history));
515:     } else {
516:       PetscCall(PetscOpenHistoryFile(NULL, &petsc_history));
517:     }
518:   }

520:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-log_sync", &PetscLogSyncOn, NULL));

522:   PetscBool flg3 = PETSC_FALSE;
523:   #if defined(PETSC_HAVE_MPE)
524:   flg1 = PETSC_FALSE;
525:   PetscCall(PetscOptionsHasName(NULL, NULL, "-log_mpe", &flg1));
526:   if (flg1) PetscCall(PetscLogMPEBegin());
527:   #endif
528:   flg1 = PETSC_FALSE;
529:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-log_all", &flg1, NULL));
530:   PetscCall(PetscOptionsHasName(NULL, NULL, "-log_summary", &flg3));
531:   if (flg1) PetscCall(PetscLogAllBegin());
532:   else if (flg3) PetscCall(PetscLogDefaultBegin());

534:   PetscCall(PetscOptionsGetString(NULL, NULL, "-log_trace", mname, sizeof(mname), &flg1));
535:   if (flg1) {
536:     char  name[PETSC_MAX_PATH_LEN], fname[PETSC_MAX_PATH_LEN];
537:     FILE *file;
