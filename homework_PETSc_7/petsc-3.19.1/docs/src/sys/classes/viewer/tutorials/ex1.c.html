  4: #include <petscviewer.h>

  8:   PetscViewer viewer;

 13:   PetscCall(PetscViewerCreate(PETSC_COMM_WORLD, &viewer));

 15:   PetscCall(PetscViewerFileSetMode(viewer, FILE_MODE_APPEND));
 16:   PetscCall(PetscViewerFileSetName(viewer, "test.txt"));
 17:   for (i = 0; i < 10; ++i) PetscCall(PetscViewerASCIIPrintf(viewer, "test line %" PetscInt_FMT "\n", i));
 18:   PetscCall(PetscViewerDestroy(&viewer));
 19:   PetscCall(PetscFinalize());
 20:   return 0;
 21: }
