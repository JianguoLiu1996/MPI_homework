 44: static inline PetscErrorCode PetscViewerVTKSanitizeName_Internal(char name[], size_t maxlen)
 45: {
 46:   size_t c;

 48:   PetscFunctionBegin;
 49:   for (c = 0; c < maxlen; c++) {
 50:     char a = name[c];
 51:     if (a == '\0') break;
 52:     if (a == ' ') name[c] = '_';
 53:   }
