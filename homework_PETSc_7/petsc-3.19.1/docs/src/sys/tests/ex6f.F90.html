 18:       PetscCallA(PetscRandomGetValue(r,rand,ierr))
 19:       print*, 'Random value:',rand
 20:       PetscCallA(PetscRandomDestroy(r,ierr))
 21:       PetscCallA(PetscFinalize(ierr))
 22:       end
