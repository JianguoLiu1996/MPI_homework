  1: static char help[] = "Test VecSetValuesCOO\n\n";

  3: #include <petscmat.h>
  4: int main(int argc, char **args)
  5: {
  6:   Vec            x, y;
  7:   PetscInt       k;
  8:   const PetscInt M = 18;
  9:   PetscMPIInt    rank, size;
 10:   PetscBool      equal;
 11:   PetscScalar   *vals;
 12:   PetscBool      ignoreRemote = PETSC_FALSE;

 14:   PetscInt i0[] = {3, 4, 1, 10, 0, 1, 1, 2, 1, 1, 2, 2, 3, 3, 4, 4, 1, 2, 5, 5, 6, 4, 17, 0, 1, 1, 8, 5, 5, 6, 4, 7, 8, 5};
 15:   PetscInt i1[] = {8, 5, 15, 16, 6, 13, 4, 17, 8, 9, 9, 10, 6, 12, 7, 3, 4, 1, 1, 2, 5, 5, 6, 14, 17, 8, 9, 9, 10, 4, 5, 10, 11, 1, 2};
 16:   PetscInt i2[] = {7, 7, 8, 8, 9, 16, 17, 9, 10, 1, 1, -2, 2, 3, 3, 14, 4, 5, 10, 13, 9, 9, 10, 1, 0, 0, 5, 5, 6, 6, 13, 13, 14, -14, 4, 4, 5, 11, 11, 12, 15, 15, 16};

 18:   struct {
 19:     PetscInt  *i;
 20:     PetscCount n;
 21:   } coo[3] = {
 22:     {i0, sizeof(i0) / sizeof(PetscInt)},
 23:     {i1, sizeof(i1) / sizeof(PetscInt)},
 24:     {i2, sizeof(i2) / sizeof(PetscInt)}
 25:   };

 27:   PetscFunctionBeginUser;
 28:   PetscCall(PetscInitialize(&argc, &args, (char *)0, help));
 29:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-ignore_remote", &ignoreRemote, NULL));
 30:   PetscCallMPI(MPI_Comm_rank(PETSC_COMM_WORLD, &rank));
 31:   PetscCallMPI(MPI_Comm_size(PETSC_COMM_WORLD, &size));

 33:   PetscCheck(size <= 3, PETSC_COMM_WORLD, PETSC_ERR_WRONG_MPI_SIZE, "This test requires at most 3 processes");

 35:   PetscCall(VecCreate(PETSC_COMM_WORLD, &x));
 36:   PetscCall(VecSetSizes(x, PETSC_DECIDE, M));
 37:   PetscCall(VecSetType(x, VECSTANDARD));
 38:   PetscCall(VecSetOption(x, VEC_IGNORE_OFF_PROC_ENTRIES, ignoreRemote));
 39:   PetscCall(VecSetOption(x, VEC_IGNORE_NEGATIVE_INDICES, PETSC_TRUE));

 41:   for (k = 0; k < coo[rank].n; k++) {
 42:     PetscScalar val = (PetscScalar)coo[rank].i[k];
 43:     PetscCall(VecSetValues(x, 1, &coo[rank].i[k], &val, ADD_VALUES));
 44:   }
 45:   PetscCall(VecAssemblyBegin(x));
 46:   PetscCall(VecAssemblyEnd(x));

 48:   PetscCall(VecCreate(PETSC_COMM_WORLD, &y));
 49:   PetscCall(VecSetSizes(y, PETSC_DECIDE, M));
 50:   PetscCall(VecSetFromOptions(y));
 51:   PetscCall(VecSetOption(y, VEC_IGNORE_OFF_PROC_ENTRIES, ignoreRemote));
 52:   PetscCall(VecSetOption(y, VEC_IGNORE_NEGATIVE_INDICES, PETSC_TRUE));
 53:   PetscCall(VecSetPreallocationCOO(y, coo[rank].n, coo[rank].i));

 55:   PetscCall(PetscMalloc1(coo[rank].n, &vals));
 56:   for (k = 0; k < coo[rank].n; k++) vals[k] = (PetscScalar)coo[rank].i[k];
 57:   PetscCall(VecSetValuesCOO(y, vals, ADD_VALUES));

 59:   PetscCall(VecEqual(x, y, &equal));

 61:   if (!equal) {
 62:     PetscCall(VecView(x, PETSC_VIEWER_STDOUT_WORLD));
 63:     PetscCall(VecView(y, PETSC_VIEWER_STDOUT_WORLD));
 64:     SETERRQ(PETSC_COMM_WORLD, PETSC_ERR_PLIB, "VecSetValuesCOO() failed");
 65:   }

 67:   PetscCall(PetscFree(vals));
 68:   PetscCall(VecDestroy(&x));
 69:   PetscCall(VecDestroy(&y));
 70:   PetscCall(PetscFinalize());
 71:   return 0;
 72: }
