  1: static char help[] = "Reads a PETSc matrix and vector from a file and solves a linear system.\n\
  2: Input arguments are:\n\
  3:   -f <input_file> : file to load. For example see $PETSC_DIR/share/petsc/datafiles/matrices\n\n";

  5: #include <petscksp.h>
  6: #include <petsclog.h>

  8: static PetscErrorCode KSPTestResidualMonitor(KSP ksp, PetscInt i, PetscReal r, void *ctx)
  9: {
 10:   Vec      *t, *v;
 11:   PetscReal err;

 13:   PetscFunctionBeginUser;
 14:   PetscCall(KSPCreateVecs(ksp, 2, &t, 2, &v));
 15:   PetscCall(KSPBuildResidualDefault(ksp, t[0], v[0], &v[0]));
 16:   PetscCall(KSPBuildResidual(ksp, t[1], v[1], &v[1]));
 17:   PetscCall(VecAXPY(v[1], -1.0, v[0]));
 18:   PetscCall(VecNorm(v[1], NORM_INFINITY, &err));
 19:   PetscCheck(err <= PETSC_SMALL, PetscObjectComm((PetscObject)ksp), PETSC_ERR_PLIB, "Inconsistent residual computed at step %" PetscInt_FMT ": %g (KSP %g)", i, (double)err, (double)r);
 20:   PetscCall(VecDestroyVecs(2, &t));
 21:   PetscCall(VecDestroyVecs(2, &v));
 22:   PetscFunctionReturn(PETSC_SUCCESS);
 23: }

 25: int main(int argc, char **args)
 26: {
 27:   PetscInt its;
 28: #if defined(PETSC_USE_LOG)
 29:   PetscLogStage stage1, stage2;
 30: #endif
 31:   PetscReal   norm;
 32:   Vec         x, b, u;
 33:   Mat         A;
 34:   char        file[PETSC_MAX_PATH_LEN];
 35:   PetscViewer fd;
 36:   PetscBool   table = PETSC_FALSE, flg, test_residual = PETSC_FALSE, b_in_f = PETSC_TRUE;
 37:   KSP         ksp;

 39:   PetscFunctionBeginUser;
 40:   PetscCall(PetscInitialize(&argc, &args, (char *)0, help));
 41:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-table", &table, NULL));
 42:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-test_residual", &test_residual, NULL));
 43:   PetscCall(PetscOptionsGetBool(NULL, NULL, "-b_in_f", &b_in_f, NULL));

 45:   /* Read matrix and RHS */
 46:   PetscCall(PetscOptionsGetString(NULL, NULL, "-f", file, sizeof(file), &flg));
 47:   PetscCheck(flg, PETSC_COMM_WORLD, PETSC_ERR_USER_INPUT, "Must indicate binary file with the -f option");
 48:   PetscCall(PetscViewerBinaryOpen(PETSC_COMM_WORLD, file, FILE_MODE_READ, &fd));
 49:   PetscCall(MatCreate(PETSC_COMM_WORLD, &A));
 50:   PetscCall(MatLoad(A, fd));
 51:   if (b_in_f) {
 52:     PetscCall(VecCreate(PETSC_COMM_WORLD, &b));
 53:     PetscCall(VecLoad(b, fd));
 54:   } else {
 55:     PetscCall(MatCreateVecs(A, NULL, &b));
 56:     PetscCall(VecSetRandom(b, NULL));
 57:   }
 58:   PetscCall(PetscViewerDestroy(&fd));

 60:   /*
 61:    If the load matrix is larger then the vector, due to being padded
 62:    to match the blocksize then create a new padded vector
 63:   */
 64:   {
 65:     PetscInt     m, n, j, mvec, start, end, indx;
 66:     Vec          tmp;
 67:     PetscScalar *bold;

 69:     PetscCall(MatGetLocalSize(A, &m, &n));
 70:     PetscCall(VecCreate(PETSC_COMM_WORLD, &tmp));
 71:     PetscCall(VecSetSizes(tmp, m, PETSC_DECIDE));
 72:     PetscCall(VecSetFromOptions(tmp));
 73:     PetscCall(VecGetOwnershipRange(b, &start, &end));
 74:     PetscCall(VecGetLocalSize(b, &mvec));
 75:     PetscCall(VecGetArray(b, &bold));
 76:     for (j = 0; j < mvec; j++) {
 77:       indx = start + j;
 78:       PetscCall(VecSetValues(tmp, 1, &indx, bold + j, INSERT_VALUES));
 79:     }
 80:     PetscCall(VecRestoreArray(b, &bold));
 81:     PetscCall(VecDestroy(&b));
 82:     PetscCall(VecAssemblyBegin(tmp));
 83:     PetscCall(VecAssemblyEnd(tmp));
 84:     b = tmp;
 85:   }
 86:   PetscCall(VecDuplicate(b, &x));
 87:   PetscCall(VecDuplicate(b, &u));

 89:   PetscCall(VecSet(x, 0.0));
 90:   PetscCall(PetscBarrier((PetscObject)A));

 92:   PetscCall(PetscLogStageRegister("mystage 1", &stage1));
 93:   PetscCall(PetscLogStagePush(stage1));
 94:   PetscCall(KSPCreate(PETSC_COMM_WORLD, &ksp));
 95:   PetscCall(KSPSetOperators(ksp, A, A));
 96:   PetscCall(KSPSetFromOptions(ksp));
 97:   if (test_residual) PetscCall(KSPMonitorSet(ksp, KSPTestResidualMonitor, NULL, NULL));
 98:   PetscCall(KSPSetUp(ksp));
 99:   PetscCall(KSPSetUpOnBlocks(ksp));
100:   PetscCall(PetscLogStagePop());
101:   PetscCall(PetscBarrier((PetscObject)A));

103:   PetscCall(PetscLogStageRegister("mystage 2", &stage2));
104:   PetscCall(PetscLogStagePush(stage2));
105:   PetscCall(KSPSolve(ksp, b, x));
106:   PetscCall(PetscLogStagePop());

108:   /* Show result */
109:   PetscCall(MatMult(A, x, u));
110:   PetscCall(VecAXPY(u, -1.0, b));
111:   PetscCall(VecNorm(u, NORM_2, &norm));
112:   PetscCall(KSPGetIterationNumber(ksp, &its));
113:   /*  matrix PC   KSP   Options       its    residual  */
114:   if (table) {
115:     char       *matrixname, kspinfo[120];
116:     PetscViewer viewer;
117:     PetscCall(PetscViewerStringOpen(PETSC_COMM_WORLD, kspinfo, sizeof(kspinfo), &viewer));
118:     PetscCall(KSPView(ksp, viewer));
119:     PetscCall(PetscStrrchr(file, '/', &matrixname));
120:     PetscCall(PetscPrintf(PETSC_COMM_WORLD, "%-8.8s %3" PetscInt_FMT " %2.0e %s \n", matrixname, its, (double)norm, kspinfo));
121:     PetscCall(PetscViewerDestroy(&viewer));
122:   } else {
123:     PetscCall(PetscPrintf(PETSC_COMM_WORLD, "Number of iterations = %3" PetscInt_FMT "\n", its));
124:     PetscCall(PetscPrintf(PETSC_COMM_WORLD, "Residual norm = %g\n", (double)norm));
125:   }

127:   /* Cleanup */
128:   PetscCall(KSPDestroy(&ksp));
129:   PetscCall(VecDestroy(&x));
130:   PetscCall(VecDestroy(&b));
131:   PetscCall(VecDestroy(&u));
132:   PetscCall(MatDestroy(&A));
133:   PetscCall(PetscFinalize());
134:   return 0;
135: }
