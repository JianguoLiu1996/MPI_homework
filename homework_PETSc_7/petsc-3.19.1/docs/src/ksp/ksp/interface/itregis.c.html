131: @*/
132: PetscErrorCode KSPMonitorRegisterAll(void)
133: {
134:   PetscFunctionBegin;

143:   PetscCall(KSPMonitorRegister("true_residual", PETSCVIEWERASCII, PETSC_VIEWER_DEFAULT, KSPMonitorTrueResidual, NULL, NULL));
