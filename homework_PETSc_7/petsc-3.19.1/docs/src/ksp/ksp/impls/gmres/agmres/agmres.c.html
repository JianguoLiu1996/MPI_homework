 23: /*
 24:    This function allocates  data for the Newton basis GMRES implementation.
 25:    Note that most data are allocated in KSPSetUp_DGMRES and KSPSetUp_GMRES, including the space for the basis vectors, the various Hessenberg matrices and the Givens rotations coefficients

 27: */
 28: static PetscErrorCode KSPSetUp_AGMRES(KSP ksp)
 29: {
 30:   PetscInt       hes;
 31:   PetscInt       nloc;
 32:   KSP_AGMRES    *agmres = (KSP_AGMRES *)ksp->data;
 33:   PetscInt       neig   = agmres->neig;
 34:   const PetscInt max_k  = agmres->max_k;
 35:   PetscInt       N      = MAXKSPSIZE;
 36:   PetscInt       lwork  = PetscMax(8 * N + 16, 4 * neig * (N - neig));

 38:   PetscFunctionBegin;
 39:   N = MAXKSPSIZE;
 40:   /* Preallocate space during the call to KSPSetup_GMRES for the Krylov basis */
 41:   agmres->q_preallocate = PETSC_TRUE; /* No allocation on the fly */
 42:   /* Preallocate space to compute later the eigenvalues in GMRES */
 43:   ksp->calc_sings = PETSC_TRUE;
 44:   agmres->max_k   = N; /* Set the augmented size to be allocated in KSPSetup_GMRES */
 45:   PetscCall(KSPSetUp_DGMRES(ksp));
 46:   agmres->max_k = max_k;
 47:   hes           = (N + 1) * (N + 1);

 49:   /* Data for the Newton basis GMRES */
 50:   PetscCall(PetscCalloc4(max_k, &agmres->Rshift, max_k, &agmres->Ishift, hes, &agmres->Rloc, (N + 1) * 4, &agmres->wbufptr));
 51:   PetscCall(PetscMalloc3(N + 1, &agmres->tau, lwork, &agmres->work, N + 1, &agmres->nrs));
 52:   PetscCall(PetscCalloc4(N + 1, &agmres->Scale, N + 1, &agmres->sgn, N + 1, &agmres->tloc, N + 1, &agmres->temp));

 54:   /* Allocate space for the vectors in the orthogonalized basis*/
 55:   PetscCall(VecGetLocalSize(agmres->vecs[0], &nloc));
 56:   PetscCall(PetscMalloc1(nloc * (N + 1), &agmres->Qloc));

 58:   /* Init the ring of processors for the roddec orthogonalization */
 59:   PetscCall(KSPAGMRESRoddecInitNeighboor(ksp));

 61:   if (agmres->neig < 1) PetscFunctionReturn(PETSC_SUCCESS);

 63:   /* Allocate space for the deflation */
 64:   PetscCall(PetscMalloc1(N, &agmres->select));
 65:   PetscCall(VecDuplicateVecs(VEC_V(0), N, &agmres->TmpU));
 66:   PetscCall(PetscMalloc2(N * N, &agmres->MatEigL, N * N, &agmres->MatEigR));
 67:   /*  PetscCall(PetscMalloc6(N*N, &agmres->Q, N*N, &agmres->Z, N, &agmres->wr, N, &agmres->wi, N, &agmres->beta, N, &agmres->modul)); */
 68:   PetscCall(PetscMalloc3(N * N, &agmres->Q, N * N, &agmres->Z, N, &agmres->beta));
 69:   PetscCall(PetscMalloc2((N + 1), &agmres->perm, (2 * neig * N), &agmres->iwork));
 70:   PetscFunctionReturn(PETSC_SUCCESS);
 71: }

 74:     Returns the current solution from the private data structure of AGMRES back to ptr.

 76: static PetscErrorCode KSPBuildSolution_AGMRES(KSP ksp, Vec ptr, Vec *result)
 77: {
 78:   KSP_AGMRES *agmres = (KSP_AGMRES *)ksp->data;

 80:   PetscFunctionBegin;
 81:   if (!ptr) {
 82:     if (!agmres->sol_temp) {
 83:       PetscCall(VecDuplicate(ksp->vec_sol, &agmres->sol_temp));
 84:       PetscCall(VecCopy(ksp->vec_sol, agmres->sol_temp));
 85:     }
 86:     ptr = agmres->sol_temp;
 87:   } else {
 88:     PetscCall(VecCopy(ksp->vec_sol, ptr));
 89:   }
 90:   if (result) *result = ptr;
 91:   PetscFunctionReturn(PETSC_SUCCESS);
 92: }

 94: /* Computes the shifts  needed to generate stable basis vectors (through the Newton polynomials)
 95:    At input, the operators (matrix and preconditioners) are used to create a new GMRES KSP.
 96:    One cycle of GMRES with the Arnoldi process is performed and the eigenvalues of the induced Hessenberg matrix (the Ritz values) are computed.
 97:    NOTE: This function is not currently used; the next function is rather used when  the eigenvectors are needed next to augment the basis
 98: */
 99: PetscErrorCode KSPComputeShifts_GMRES(KSP ksp)
100: {
101:   KSP_AGMRES    *agmres = (KSP_AGMRES *)(ksp->data);
102:   KSP            kspgmres;
103:   Mat            Amat, Pmat;
104:   const PetscInt max_k = agmres->max_k;
105:   PC             pc;
106:   PetscInt       m;
107:   PetscScalar   *Rshift, *Ishift;
108:   PetscBool      flg;

110:   PetscFunctionBegin;
111:   /* Perform one cycle of classical GMRES (with the Arnoldi process) to get the Hessenberg matrix
112:    We assume here that the ksp is AGMRES and that the operators for the
113:    linear system have been set in this ksp */
114:   PetscCall(KSPCreate(PetscObjectComm((PetscObject)ksp), &kspgmres));
115:   if (!ksp->pc) PetscCall(KSPGetPC(ksp, &ksp->pc));
116:   PetscCall(PCGetOperators(ksp->pc, &Amat, &Pmat));
117:   PetscCall(KSPSetOperators(kspgmres, Amat, Pmat));
118:   PetscCall(KSPSetFromOptions(kspgmres));
119:   PetscCall(PetscOptionsHasName(NULL, ((PetscObject)ksp)->prefix, "-ksp_view", &flg));
120:   if (flg) PetscCall(PetscOptionsClearValue(NULL, "-ksp_view"));
121:   PetscCall(KSPSetType(kspgmres, KSPGMRES));
122:   PetscCall(KSPGMRESSetRestart(kspgmres, max_k));
123:   PetscCall(KSPGetPC(ksp, &pc));
124:   PetscCall(KSPSetPC(kspgmres, pc));
125:   /* Copy common options */
126:   kspgmres->pc_side = ksp->pc_side;
127:   /* Setup KSP context */
128:   PetscCall(KSPSetComputeEigenvalues(kspgmres, PETSC_TRUE));
129:   PetscCall(KSPSetUp(kspgmres));

131:   kspgmres->max_it = max_k; /* Restrict the maximum number of iterations to one cycle of GMRES */
132:   kspgmres->rtol   = ksp->rtol;

134:   PetscCall(KSPSolve(kspgmres, ksp->vec_rhs, ksp->vec_sol));

136:   ksp->guess_zero = PETSC_FALSE;
137:   ksp->rnorm      = kspgmres->rnorm;
138:   ksp->its        = kspgmres->its;
139:   if (kspgmres->reason == KSP_CONVERGED_RTOL) {
140:     ksp->reason = KSP_CONVERGED_RTOL;
141:     PetscFunctionReturn(PETSC_SUCCESS);
142:   } else ksp->reason = KSP_CONVERGED_ITERATING;
143:   /* Now, compute the Shifts values */
144:   PetscCall(PetscMalloc2(max_k, &Rshift, max_k, &Ishift));
145:   PetscCall(KSPComputeEigenvalues(kspgmres, max_k, Rshift, Ishift, &m));
146:   PetscCheck(m >= max_k, PetscObjectComm((PetscObject)ksp), PETSC_ERR_PLIB, "Unable to compute the Shifts for the Newton basis");
147:   PetscCall(KSPAGMRESLejaOrdering(Rshift, Ishift, agmres->Rshift, agmres->Ishift, max_k));

149:   agmres->HasShifts = PETSC_TRUE;
150:   /* Restore KSP view options */
151:   if (flg) PetscCall(PetscOptionsSetValue(NULL, "-ksp_view", ""));
152:   PetscFunctionReturn(PETSC_SUCCESS);
153: }

155: /* Computes the shift values (Ritz values) needed to generate stable basis vectors
156:    One cycle of DGMRES is performed to find the eigenvalues. The same data structures are used since AGMRES extends DGMRES
157:    Note that when the basis is  to be augmented, then this function computes the harmonic Ritz vectors from this first cycle.
158:    Input :
159:     - The operators (matrix, preconditioners and right hand side) are  normally required.
160:     - max_k : the size of the (non augmented) basis.
161:     - neig: The number of eigenvectors to augment, if deflation is needed
162:    Output :
163:     - The shifts as complex pair of arrays in wr and wi (size max_k).
164:     - The harmonic Ritz vectors (agmres->U) if deflation is needed.
165: */
166: static PetscErrorCode KSPComputeShifts_DGMRES(KSP ksp)
167: {
168:   KSP_AGMRES    *agmres = (KSP_AGMRES *)(ksp->data);
169:   PetscInt       max_k  = agmres->max_k; /* size of the (non augmented) Krylov subspace */
170:   PetscInt       Neig   = 0;
171:   const PetscInt max_it = ksp->max_it;
172:   PetscBool      flg;

174:   /* Perform one cycle of dgmres to find the eigenvalues and compute the first approximations of the eigenvectors */

176:   PetscFunctionBegin;
177:   PetscCall(PetscLogEventBegin(KSP_AGMRESComputeShifts, ksp, 0, 0, 0));
178:   /* Send the size of the augmented basis to DGMRES */
179:   ksp->max_it             = max_k; /* set this to have DGMRES performing only one cycle */
180:   ksp->ops->buildsolution = KSPBuildSolution_DGMRES;
181:   PetscCall(KSPSolve_DGMRES(ksp));
182:   ksp->guess_zero = PETSC_FALSE;
183:   if (ksp->reason == KSP_CONVERGED_RTOL) {
184:     PetscCall(PetscLogEventEnd(KSP_AGMRESComputeShifts, ksp, 0, 0, 0));
185:     PetscFunctionReturn(PETSC_SUCCESS);
186:   } else ksp->reason = KSP_CONVERGED_ITERATING;

188:   if ((agmres->r == 0) && (agmres->neig > 0)) { /* Compute the eigenvalues for the shifts and the eigenvectors (to augment the Newton basis) */
189:     agmres->HasSchur = PETSC_FALSE;
190:     PetscCall(KSPDGMRESComputeDeflationData_DGMRES(ksp, &Neig));
191:     Neig = max_k;
192:   } else { /* From DGMRES, compute only the eigenvalues needed as Shifts for the Newton Basis */
193:     PetscCall(KSPDGMRESComputeSchurForm_DGMRES(ksp, &Neig));
194:   }

196:   /* It may happen that the Ritz values from one cycle of GMRES are not accurate enough to provide a good stability. In this case, another cycle of GMRES is performed.  The two sets of values thus generated are sorted and the most accurate are kept as shifts */
197:   PetscCall(PetscOptionsHasName(NULL, NULL, "-ksp_agmres_ImproveShifts", &flg));
198:   if (!flg) {
199:     PetscCall(KSPAGMRESLejaOrdering(agmres->wr, agmres->wi, agmres->Rshift, agmres->Ishift, max_k));
200:   } else { /* Perform another cycle of DGMRES to find another set of eigenvalues */
201:     PetscInt     i;
202:     PetscScalar *wr, *wi, *Rshift, *Ishift;
203:     PetscCall(PetscMalloc4(2 * max_k, &wr, 2 * max_k, &wi, 2 * max_k, &Rshift, 2 * max_k, &Ishift));
204:     for (i = 0; i < max_k; i++) {
205:       wr[i] = agmres->wr[i];
206:       wi[i] = agmres->wi[i];
207:     }

209:     PetscCall(KSPSolve_DGMRES(ksp));

211:     ksp->guess_zero = PETSC_FALSE;
212:     if (ksp->reason == KSP_CONVERGED_RTOL) PetscFunctionReturn(PETSC_SUCCESS);
213:     else ksp->reason = KSP_CONVERGED_ITERATING;
214:     if (agmres->neig > 0) { /* Compute the eigenvalues for the shifts) and the eigenvectors (to augment the Newton basis */
215:       agmres->HasSchur = PETSC_FALSE;

217:       PetscCall(KSPDGMRESComputeDeflationData_DGMRES(ksp, &Neig));
218:       Neig = max_k;
219:     } else { /* From DGMRES, compute only the eigenvalues needed as Shifts for the Newton Basis */
220:       PetscCall(KSPDGMRESComputeSchurForm_DGMRES(ksp, &Neig));
221:     }
222:     for (i = 0; i < max_k; i++) {
223:       wr[max_k + i] = agmres->wr[i];
224:       wi[max_k + i] = agmres->wi[i];
225:     }
226:     PetscCall(KSPAGMRESLejaOrdering(wr, wi, Rshift, Ishift, 2 * max_k));
227:     for (i = 0; i < max_k; i++) {
228:       agmres->Rshift[i] = Rshift[i];
229:       agmres->Ishift[i] = Ishift[i];
230:     }
231:     PetscCall(PetscFree(Rshift));
232:     PetscCall(PetscFree(wr));
233:     PetscCall(PetscFree(Ishift));
234:     PetscCall(PetscFree(wi));
235:   }
236:   agmres->HasShifts = PETSC_TRUE;
237:   ksp->max_it       = max_it;
238:   PetscCall(PetscLogEventEnd(KSP_AGMRESComputeShifts, ksp, 0, 0, 0));
239:   PetscFunctionReturn(PETSC_SUCCESS);
240: }

242: /*
243:    Generate the basis vectors from the Newton polynomials with shifts and scaling factors
244:    The scaling factors are computed to obtain unit vectors. Note that this step can be avoided with the preprocessing option KSP_AGMRES_NONORM.
245:    Inputs :
246:     - Operators (Matrix and preconditioners and the first basis vector in VEC_V(0)
247:     - Shifts values in agmres->Rshift and agmres->Ishift.
248:    Output :
249:     - agmres->vecs or VEC_V : basis vectors
250:     - agmres->Scale : Scaling factors (equal to 1 if no scaling is done)
251: */
252: static PetscErrorCode KSPAGMRESBuildBasis(KSP ksp)
253: {
254:   KSP_AGMRES    *agmres  = (KSP_AGMRES *)ksp->data;
255:   PetscReal     *Rshift  = agmres->Rshift;
256:   PetscReal     *Ishift  = agmres->Ishift;
257:   PetscReal     *Scale   = agmres->Scale;
258:   const PetscInt max_k   = agmres->max_k;
259:   PetscInt       KspSize = KSPSIZE; /* if max_k == KspSizen then the basis should not be augmented */
260:   PetscInt       j       = 1;

262:   PetscFunctionBegin;
263:   PetscCall(PetscLogEventBegin(KSP_AGMRESBuildBasis, ksp, 0, 0, 0));
264:   Scale[0] = 1.0;
265:   while (j <= max_k) {
266:     if (Ishift[j - 1] == 0) {
267:       if ((ksp->pc_side == PC_LEFT) && agmres->r && agmres->DeflPrecond) {
268:         /* Apply the precond-matrix operators */
269:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_V(j - 1), VEC_TMP, VEC_TMP_MATOP));
270:         /* Then apply deflation as a preconditioner */
271:         PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_TMP, VEC_V(j)));
272:       } else if ((ksp->pc_side == PC_RIGHT) && agmres->r && agmres->DeflPrecond) {
273:         PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_V(j - 1), VEC_TMP));
274:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_TMP, VEC_V(j), VEC_TMP_MATOP));
275:       } else {
276:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_V(j - 1), VEC_V(j), VEC_TMP_MATOP));
277:       }
278:       PetscCall(VecAXPY(VEC_V(j), -Rshift[j - 1], VEC_V(j - 1)));
279: #if defined(KSP_AGMRES_NONORM)
280:       Scale[j] = 1.0;
281: #else
282:       PetscCall(VecScale(VEC_V(j), Scale[j - 1])); /* This step can be postponed until all vectors are built */
283:       PetscCall(VecNorm(VEC_V(j), NORM_2, &(Scale[j])));
284:       Scale[j] = 1.0 / Scale[j];
285: #endif

287:       agmres->matvecs += 1;
288:       j++;
289:     } else {
290:       if ((ksp->pc_side == PC_LEFT) && agmres->r && agmres->DeflPrecond) {
291:         /* Apply the precond-matrix operators */
292:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_V(j - 1), VEC_TMP, VEC_TMP_MATOP));
293:         /* Then apply deflation as a preconditioner */
294:         PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_TMP, VEC_V(j)));
295:       } else if ((ksp->pc_side == PC_RIGHT) && agmres->r && agmres->DeflPrecond) {
296:         PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_V(j - 1), VEC_TMP));
297:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_TMP, VEC_V(j), VEC_TMP_MATOP));
298:       } else {
299:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_V(j - 1), VEC_V(j), VEC_TMP_MATOP));
300:       }
301:       PetscCall(VecAXPY(VEC_V(j), -Rshift[j - 1], VEC_V(j - 1)));
302: #if defined(KSP_AGMRES_NONORM)
303:       Scale[j] = 1.0;
304: #else
305:       PetscCall(VecScale(VEC_V(j), Scale[j - 1]));
306:       PetscCall(VecNorm(VEC_V(j), NORM_2, &(Scale[j])));
307:       Scale[j] = 1.0 / Scale[j];
308: #endif
309:       agmres->matvecs += 1;
310:       j++;
311:       if ((ksp->pc_side == PC_LEFT) && agmres->r && agmres->DeflPrecond) {
312:         /* Apply the precond-matrix operators */
313:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_V(j - 1), VEC_TMP, VEC_TMP_MATOP));
314:         /* Then apply deflation as a preconditioner */
315:         PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_TMP, VEC_V(j)));
316:       } else if ((ksp->pc_side == PC_RIGHT) && agmres->r && agmres->DeflPrecond) {
317:         PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_V(j - 1), VEC_TMP));
318:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_TMP, VEC_V(j), VEC_TMP_MATOP));
319:       } else {
320:         PetscCall(KSP_PCApplyBAorAB(ksp, VEC_V(j - 1), VEC_V(j), VEC_TMP_MATOP));
321:       }
322:       PetscCall(VecAXPY(VEC_V(j), -Rshift[j - 2], VEC_V(j - 1)));
323:       PetscCall(VecAXPY(VEC_V(j), Scale[j - 2] * Ishift[j - 2] * Ishift[j - 2], VEC_V(j - 2)));
324: #if defined(KSP_AGMRES_NONORM)
325:       Scale[j] = 1.0;
326: #else
327:       PetscCall(VecNorm(VEC_V(j), NORM_2, &(Scale[j])));
328:       Scale[j] = 1.0 / Scale[j];
329: #endif
330:       agmres->matvecs += 1;
331:       j++;
332:     }
333:   }
334:   /* Augment the subspace with the eigenvectors*/
335:   while (j <= KspSize) {
336:     PetscCall(KSP_PCApplyBAorAB(ksp, agmres->U[j - max_k - 1], VEC_V(j), VEC_TMP_MATOP));
337: #if defined(KSP_AGMRES_NONORM)
338:     Scale[j] = 1.0;
339: #else
340:     PetscCall(VecScale(VEC_V(j), Scale[j - 1]));
341:     PetscCall(VecNorm(VEC_V(j), NORM_2, &(Scale[j])));
342:     Scale[j] = 1.0 / Scale[j];
343: #endif
344:     agmres->matvecs += 1;
345:     j++;
346:   }
347:   PetscCall(PetscLogEventEnd(KSP_AGMRESBuildBasis, ksp, 0, 0, 0));
348:   PetscFunctionReturn(PETSC_SUCCESS);
349: }

351: /*
352:   Form the Hessenberg matrix for the Arnoldi-like relation.
353:    Inputs :
354:    - Shifts values in agmres->Rshift and agmres->Ishift
355:    - RLoc : Triangular matrix from the RODDEC orthogonalization
356:    Outputs :
357:    - H = agmres->hh_origin : The Hessenberg matrix.

359:    NOTE: Note that the computed Hessenberg matrix is not mathematically equivalent to that in the real Arnoldi process (in KSP GMRES). If it is needed, it can be explicitly  formed as H <-- H * RLoc^-1.
360:  */
361: static PetscErrorCode KSPAGMRESBuildHessenberg(KSP ksp)
362: {
363:   KSP_AGMRES    *agmres = (KSP_AGMRES *)ksp->data;
364:   PetscScalar   *Rshift = agmres->Rshift;
365:   PetscScalar   *Ishift = agmres->Ishift;
366:   PetscScalar   *Scale  = agmres->Scale;
367:   PetscInt       i = 0, j = 0;
368:   const PetscInt max_k   = agmres->max_k;
369:   PetscInt       KspSize = KSPSIZE;
370:   PetscInt       N       = MAXKSPSIZE + 1;

372:   PetscFunctionBegin;
373:   PetscCall(PetscArrayzero(agmres->hh_origin, (N + 1) * N));
374:   while (j < max_k) {
375:     /* Real shifts */
376:     if (Ishift[j] == 0) {
377:       for (i = 0; i <= j; i++) *H(i, j) = *RLOC(i, j + 1) / Scale[j] + (Rshift[j] * *RLOC(i, j));
378:       *H(j + 1, j) = *RLOC(j + 1, j + 1) / Scale[j];
379:       j++;
380:     } else if (Ishift[j] > 0) {
381:       for (i = 0; i <= j; i++) *H(i, j) = *RLOC(i, j + 1) / Scale[j] + Rshift[j] * *RLOC(i, j);
382:       *H(j + 1, j) = *RLOC(j + 1, j + 1) / Scale[j];
383:       j++;
384:       for (i = 0; i <= j; i++) *H(i, j) = (*RLOC(i, j + 1) + Rshift[j - 1] * *RLOC(i, j) - Scale[j - 1] * Ishift[j - 1] * Ishift[j - 1] * *RLOC(i, j - 1));
385:       *H(j, j)     = (*RLOC(j, j + 1) + Rshift[j - 1] * *RLOC(j, j));
386:       *H(j + 1, j) = *RLOC(j + 1, j + 1);
387:       j++;
388:     } else SETERRQ(PetscObjectComm((PetscObject)ksp), PETSC_ERR_ORDER, "BAD ORDERING OF THE SHIFTS VALUES IN THE NEWTON BASIS");
389:   }
390:   for (j = max_k; j < KspSize; j++) { /* take into account the norm of the augmented vectors */
391:     for (i = 0; i <= j + 1; i++) *H(i, j) = *RLOC(i, j + 1) / Scale[j];
392:   }
393:   PetscFunctionReturn(PETSC_SUCCESS);
394: }

396: /*
397:   Form the new approximate solution from the least-square problem
398: */
399: static PetscErrorCode KSPAGMRESBuildSoln(KSP ksp, PetscInt it)
400: {
401:   KSP_AGMRES    *agmres = (KSP_AGMRES *)ksp->data;
402:   const PetscInt max_k  = agmres->max_k; /* Size of the non-augmented Krylov basis */
403:   PetscInt       i, j;
404:   PetscInt       r = agmres->r; /* current number of augmented eigenvectors */
405:   PetscBLASInt   KspSize;
406:   PetscBLASInt   lC;
407:   PetscBLASInt   N;
408:   PetscBLASInt   ldH = it + 1;
409:   PetscBLASInt   lwork;
410:   PetscBLASInt   info, nrhs = 1;

412:   PetscFunctionBegin;
413:   PetscCall(PetscBLASIntCast(KSPSIZE, &KspSize));
414:   PetscCall(PetscBLASIntCast(4 * (KspSize + 1), &lwork));
415:   PetscCall(PetscBLASIntCast(KspSize + 1, &lC));
416:   PetscCall(PetscBLASIntCast(MAXKSPSIZE + 1, &N));
417:   PetscCall(PetscBLASIntCast(N + 1, &ldH));
418:   /* Save a copy of the Hessenberg matrix */
419:   for (j = 0; j < N - 1; j++) {
420:     for (i = 0; i < N; i++) *HS(i, j) = *H(i, j);
421:   }
422:   /* QR factorize the Hessenberg matrix */
423:   PetscCallBLAS("LAPACKgeqrf", LAPACKgeqrf_(&lC, &KspSize, agmres->hh_origin, &ldH, agmres->tau, agmres->work, &lwork, &info));
424:   PetscCheck(!info, PetscObjectComm((PetscObject)ksp), PETSC_ERR_LIB, "Error in LAPACK routine XGEQRF INFO=%" PetscBLASInt_FMT, info);
425:   /* Update the right hand side of the least square problem */
426:   PetscCall(PetscArrayzero(agmres->nrs, N));

428:   agmres->nrs[0] = ksp->rnorm;
429:   PetscCallBLAS("LAPACKormqr", LAPACKormqr_("L", "T", &lC, &nrhs, &KspSize, agmres->hh_origin, &ldH, agmres->tau, agmres->nrs, &N, agmres->work, &lwork, &info));
430:   PetscCheck(!info, PetscObjectComm((PetscObject)ksp), PETSC_ERR_LIB, "Error in LAPACK routine XORMQR INFO=%" PetscBLASInt_FMT, info);
431:   ksp->rnorm = PetscAbsScalar(agmres->nrs[KspSize]);
432:   /* solve the least-square problem */
433:   PetscCallBLAS("LAPACKtrtrs", LAPACKtrtrs_("U", "N", "N", &KspSize, &nrhs, agmres->hh_origin, &ldH, agmres->nrs, &N, &info));
434:   PetscCheck(!info, PetscObjectComm((PetscObject)ksp), PETSC_ERR_LIB, "Error in LAPACK routine XTRTRS INFO=%" PetscBLASInt_FMT, info);
435:   /* Accumulate the correction to the solution of the preconditioned problem in VEC_TMP */
436:   PetscCall(VecZeroEntries(VEC_TMP));
437:   PetscCall(VecMAXPY(VEC_TMP, max_k, agmres->nrs, &VEC_V(0)));
438:   if (!agmres->DeflPrecond) PetscCall(VecMAXPY(VEC_TMP, r, &agmres->nrs[max_k], agmres->U));

440:   if ((ksp->pc_side == PC_RIGHT) && agmres->r && agmres->DeflPrecond) {
441:     PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_TMP, VEC_TMP_MATOP));
442:     PetscCall(VecCopy(VEC_TMP_MATOP, VEC_TMP));
443:   }
444:   PetscCall(KSPUnwindPreconditioner(ksp, VEC_TMP, VEC_TMP_MATOP));
445:   /* add the solution to the previous one */
446:   PetscCall(VecAXPY(ksp->vec_sol, 1.0, VEC_TMP));
447:   PetscFunctionReturn(PETSC_SUCCESS);
448: }

450: /*
451:    Run  one cycle of the Newton-basis gmres, possibly augmented with eigenvectors.

453:    Return residual history if requested.
454:    Input :
455:    - The vector VEC_V(0) is the initia residual
456:    Output :
457:     - the solution vector is in agmres->vec_sol
458:    - itcount : number of inner iterations
459:     - res : the new residual norm
460:  .
461:  NOTE: Unlike GMRES where the residual norm is available at each (inner) iteration,  here it is available at the end of the cycle.
462: */
463: static PetscErrorCode KSPAGMRESCycle(PetscInt *itcount, KSP ksp)
464: {
465:   KSP_AGMRES *agmres = (KSP_AGMRES *)(ksp->data);
466:   PetscReal   res;
467:   PetscInt    KspSize = KSPSIZE;

469:   PetscFunctionBegin;
470:   /* check for the convergence */
471:   res = ksp->rnorm; /* Norm of the initial residual vector */
472:   if (!res) {
473:     if (itcount) *itcount = 0;
474:     ksp->reason = KSP_CONVERGED_ATOL;
475:     PetscCall(PetscInfo(ksp, "Converged due to zero residual norm on entry\n"));
476:     PetscFunctionReturn(PETSC_SUCCESS);
477:   }
478:   PetscCall((*ksp->converged)(ksp, ksp->its, res, &ksp->reason, ksp->cnvP));
479:   /* Build the Krylov basis with Newton polynomials */
480:   PetscCall(KSPAGMRESBuildBasis(ksp));
481:   /* QR Factorize the basis with RODDEC */
482:   PetscCall(KSPAGMRESRoddec(ksp, KspSize + 1));

484:   /* Recover a (partial) Hessenberg matrix for the Arnoldi-like relation */
485:   PetscCall(KSPAGMRESBuildHessenberg(ksp));
486:   /* Solve the least square problem and unwind the preconditioner */
487:   PetscCall(KSPAGMRESBuildSoln(ksp, KspSize));

489:   res = ksp->rnorm;
490:   ksp->its += KspSize;
491:   agmres->it = KspSize - 1;
492:   /*  Test for the convergence */
493:   PetscCall((*ksp->converged)(ksp, ksp->its, res, &ksp->reason, ksp->cnvP));
494:   PetscCall(KSPLogResidualHistory(ksp, res));
495:   PetscCall(KSPMonitor(ksp, ksp->its, res));

497:   *itcount = KspSize;
498:   PetscFunctionReturn(PETSC_SUCCESS);
499: }

501: static PetscErrorCode KSPSolve_AGMRES(KSP ksp)
502: {
503:   PetscInt    its;
504:   KSP_AGMRES *agmres     = (KSP_AGMRES *)ksp->data;
505:   PetscBool   guess_zero = ksp->guess_zero;
506:   PetscReal   res_old, res;
507:   PetscInt    test;

509:   PetscFunctionBegin;
510:   PetscCall(PetscObjectSAWsTakeAccess((PetscObject)ksp));
511:   ksp->its = 0;
512:   PetscCall(PetscObjectSAWsGrantAccess((PetscObject)ksp));

514:   if (!agmres->HasShifts) { /* Compute Shifts for the Newton basis */
515:     PetscCall(KSPComputeShifts_DGMRES(ksp));
516:   }
517:   /* NOTE: At this step, the initial guess is not equal to zero since one cycle of the classical GMRES is performed to compute the shifts */
518:   PetscCall((*ksp->converged)(ksp, 0, ksp->rnorm, &ksp->reason, ksp->cnvP));
519:   while (!ksp->reason) {
520:     PetscCall(KSPInitialResidual(ksp, ksp->vec_sol, VEC_TMP, VEC_TMP_MATOP, VEC_V(0), ksp->vec_rhs));
521:     if ((ksp->pc_side == PC_LEFT) && agmres->r && agmres->DeflPrecond) {
522:       PetscCall(KSPDGMRESApplyDeflation_DGMRES(ksp, VEC_V(0), VEC_TMP));
523:       PetscCall(VecCopy(VEC_TMP, VEC_V(0)));

525:       agmres->matvecs += 1;
526:     }
527:     PetscCall(VecNormalize(VEC_V(0), &(ksp->rnorm)));
528:     KSPCheckNorm(ksp, ksp->rnorm);
529:     res_old = ksp->rnorm; /* Record the residual norm to test if deflation is needed */

531:     ksp->ops->buildsolution = KSPBuildSolution_AGMRES;

533:     PetscCall(KSPAGMRESCycle(&its, ksp));
534:     if (ksp->its >= ksp->max_it) {
535:       if (!ksp->reason) ksp->reason = KSP_DIVERGED_ITS;
536:       break;
537:     }
538:     /* compute the eigenvectors to augment the subspace : use an adaptive strategy */
539:     res = ksp->rnorm;
540:     if (!ksp->reason && agmres->neig > 0) {
541:       test = agmres->max_k * PetscLogReal(ksp->rtol / res) / PetscLogReal(res / res_old); /* estimate the remaining number of steps */
542:       if ((test > agmres->smv * (ksp->max_it - ksp->its)) || agmres->force) {
543:         if (!agmres->force && ((test > agmres->bgv * (ksp->max_it - ksp->its)) && ((agmres->r + 1) < agmres->max_neig))) { agmres->neig += 1; /* Augment the number of eigenvalues to deflate if the convergence is too slow */ }
544:         PetscCall(KSPDGMRESComputeDeflationData_DGMRES(ksp, &agmres->neig));
545:       }
546:     }
547:     ksp->guess_zero = PETSC_FALSE; /* every future call to KSPInitialResidual() will have nonzero guess */
548:   }
549:   ksp->guess_zero = guess_zero; /* restore if user has provided nonzero initial guess */
550:   PetscFunctionReturn(PETSC_SUCCESS);
551: }

553: static PetscErrorCode KSPDestroy_AGMRES(KSP ksp)
554: {
555:   KSP_AGMRES *agmres = (KSP_AGMRES *)ksp->data;

557:   PetscFunctionBegin;
558:   PetscCall(PetscFree(agmres->hh_origin));

560:   PetscCall(PetscFree(agmres->Qloc));
561:   PetscCall(PetscFree4(agmres->Rshift, agmres->Ishift, agmres->Rloc, agmres->wbufptr));
562:   PetscCall(PetscFree3(agmres->tau, agmres->work, agmres->nrs));
563:   PetscCall(PetscFree4(agmres->Scale, agmres->sgn, agmres->tloc, agmres->temp));

565:   PetscCall(PetscFree(agmres->select));
566:   PetscCall(PetscFree(agmres->wr));
567:   PetscCall(PetscFree(agmres->wi));
568:   if (agmres->neig) {
569:     PetscCall(VecDestroyVecs(MAXKSPSIZE, &agmres->TmpU));
570:     PetscCall(PetscFree(agmres->perm));
571:     PetscCall(PetscFree(agmres->MatEigL));
572:     PetscCall(PetscFree(agmres->MatEigR));
573:     PetscCall(PetscFree(agmres->Q));
574:     PetscCall(PetscFree(agmres->Z));
575:     PetscCall(PetscFree(agmres->beta));
576:   }
577:   PetscCall(KSPDestroy_DGMRES(ksp));
578:   PetscFunctionReturn(PETSC_SUCCESS);
579: }
