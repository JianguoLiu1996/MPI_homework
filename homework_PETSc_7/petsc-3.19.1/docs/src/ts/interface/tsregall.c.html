 26: /*@C
 27:   TSRegisterAll - Registers all of the timesteppers in the `TS` package.

 29:   Not Collective

 31:   Input parameter:
 32: . path - The dynamic library path

 34:   Level: advanced

 36: .seealso: [](chapter_ts), `TS`, `TSCreate()`, `TSRegister()`, `TSRegisterDestroy()`
 37: @*/
 38: PetscErrorCode TSRegisterAll(void)
 39: {
 40:   PetscFunctionBegin;
 41:   if (TSRegisterAllCalled) PetscFunctionReturn(PETSC_SUCCESS);
 42:   TSRegisterAllCalled = PETSC_TRUE;

 44:   PetscCall(TSRegister(TSEULER, TSCreate_Euler));
 45:   PetscCall(TSRegister(TSBEULER, TSCreate_BEuler));
 46:   PetscCall(TSRegister(TSCN, TSCreate_CN));
 47:   PetscCall(TSRegister(TSPSEUDO, TSCreate_Pseudo));
 48:   PetscCall(TSRegister(TSGLLE, TSCreate_GLLE));
 49:   PetscCall(TSRegister(TSSSP, TSCreate_SSP));
 50:   PetscCall(TSRegister(TSTHETA, TSCreate_Theta));
 51:   PetscCall(TSRegister(TSALPHA, TSCreate_Alpha));
 52:   PetscCall(TSRegister(TSALPHA2, TSCreate_Alpha2));
 53: #if defined(PETSC_HAVE_SUNDIALS2)
 54:   PetscCall(TSRegister(TSSUNDIALS, TSCreate_Sundials));
 55: #endif
 56: #if defined(PETSC_HAVE_RADAU5)
 57:   PetscCall(TSRegister(TSRADAU5, TSCreate_Radau5));
 58: #endif
 59:   PetscCall(TSRegister(TSRK, TSCreate_RK));
 60:   PetscCall(TSRegister(TSGLEE, TSCreate_GLEE));
 61:   PetscCall(TSRegister(TSARKIMEX, TSCreate_ARKIMEX));
 62:   PetscCall(TSRegister(TSROSW, TSCreate_RosW));
 63:   PetscCall(TSRegister(TSEIMEX, TSCreate_EIMEX));
 64:   PetscCall(TSRegister(TSMIMEX, TSCreate_Mimex));
 65:   PetscCall(TSRegister(TSBDF, TSCreate_BDF));
 66:   PetscCall(TSRegister(TSBASICSYMPLECTIC, TSCreate_BasicSymplectic));
 67:   PetscCall(TSRegister(TSMPRK, TSCreate_MPRK));
 68:   PetscCall(TSRegister(TSDISCGRAD, TSCreate_DiscGrad));
 69:   PetscCall(TSRegister(TSIRK, TSCreate_IRK));
 70:   PetscFunctionReturn(PETSC_SUCCESS);
 71: }
