1054: static PetscErrorCode phi(PetscInt dim, PetscReal time, const PetscReal x[], PetscInt Nc, PetscScalar *u, void *ctx)
1055: {
1056:   PetscInt d;
1057:   u[0] = -x[1];
1058:   u[1] = x[0];
1059:   for (d = 2; d < dim; ++d) u[d] = x[d];

1063: /* \lambda \cdot (\psi_u^- - \psi_u^+) */
1064: static void f0_bd_u(PetscInt dim, PetscInt Nf, PetscInt NfAux, const PetscInt uOff[], const PetscInt uOff_x[], const PetscScalar u[], const PetscScalar u_t[], const PetscScalar u_x[], const PetscInt aOff[], const PetscInt aOff_x[], const PetscScalar a[], const PetscScalar a_t[], const PetscScalar a_x[], PetscReal t, const PetscReal x[], const PetscReal n[], PetscInt numConstants, const PetscScalar constants[], PetscScalar f0[])
1065: {
1066:   const PetscInt Nc = uOff[1] - uOff[0];
1067:   PetscInt       c;
1068:   for (c = 0; c < Nc; ++c) {
1069:     f0[c]      = u[Nc * 2 + c] + x[Nc - c - 1];
1070:     f0[Nc + c] = -(u[Nc * 2 + c] + x[Nc - c - 1]);
1071:   }
1072: }

1074: /* (d - u^+ + u^-) \cdot \psi_\lambda */
1075: static void f0_bd_l(PetscInt dim, PetscInt Nf, PetscInt NfAux, const PetscInt uOff[], const PetscInt uOff_x[], const PetscScalar u[], const PetscScalar u_t[], const PetscScalar u_x[], const PetscInt aOff[], const PetscInt aOff_x[], const PetscScalar a[], const PetscScalar a_t[], const PetscScalar a_x[], PetscReal t, const PetscReal x[], const PetscReal n[], PetscInt numConstants, const PetscScalar constants[], PetscScalar f0[])
1076: {
1077:   const PetscInt Nc = uOff[2] - uOff[1];
1078:   PetscInt       c;

1080:   for (c = 0; c < Nc; ++c) f0[c] = (c > 0 ? 1.0 : 0.0) + u[c] - u[Nc + c];
1081: }

1083: /* \psi_lambda \cdot (\psi_u^- - \psi_u^+) */
1084: static void g0_bd_ul(PetscInt dim, PetscInt Nf, PetscInt NfAux, const PetscInt uOff[], const PetscInt uOff_x[], const PetscScalar u[], const PetscScalar u_t[], const PetscScalar u_x[], const PetscInt aOff[], const PetscInt aOff_x[], const PetscScalar a[], const PetscScalar a_t[], const PetscScalar a_x[], PetscReal t, PetscReal u_tShift, const PetscReal x[], const PetscReal n[], PetscInt numConstants, const PetscScalar constants[], PetscScalar g0[])
1085: {
1086:   const PetscInt Nc = uOff[1] - uOff[0];
1087:   PetscInt       c;

1089:   for (c = 0; c < Nc; ++c) {
1090:     g0[(0 + c) * Nc + c]  = 1.0;
1091:     g0[(Nc + c) * Nc + c] = -1.0;
1092:   }
1093: }

1095: /* (-\psi_u^+ + \psi_u^-) \cdot \psi_\lambda */
1096: static void g0_bd_lu(PetscInt dim, PetscInt Nf, PetscInt NfAux, const PetscInt uOff[], const PetscInt uOff_x[], const PetscScalar u[], const PetscScalar u_t[], const PetscScalar u_x[], const PetscInt aOff[], const PetscInt aOff_x[], const PetscScalar a[], const PetscScalar a_t[], const PetscScalar a_x[], PetscReal t, PetscReal u_tShift, const PetscReal x[], const PetscReal n[], PetscInt numConstants, const PetscScalar constants[], PetscScalar g0[])
1097: {
1098:   const PetscInt Nc = uOff[2] - uOff[1];
1099:   PetscInt       c;

1101:   for (c = 0; c < Nc; ++c) {
1102:     g0[c * Nc * 2 + c]      = 1.0;
1103:     g0[c * Nc * 2 + Nc + c] = -1.0;
1104:   }
1105: }

1107: static PetscErrorCode TestAssembly(DM dm, AppCtx *user)
1108: {
1109:   Mat           J;
1110:   Vec           locX, locF;

1118:   PetscMPIInt rank;

1132:   PetscCall(PetscObjectSetName((PetscObject)J, "Jacobian"));

1134:   /* The initial guess has displacement shifted by one unit in each fault parallel direction across the fault */

1137:   initialGuess[0] = r;

1163:   keys[0].label = NULL;

1165:   keys[0].field = 0;

1168:   keys[1].value = 2;

1177:   PetscCall(VecViewFromOptions(locF, NULL, "-local_residual_view"));

1194:     const PetscInt *cone;
1195:     PetscInt        Na, a;

1205:   PetscFunctionReturn(PETSC_SUCCESS);
1206: }
