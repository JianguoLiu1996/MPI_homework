  2: static char help[] = "Tests DMLocalToGlobal() for dof > 1\n\n";

  4: #include <petscdm.h>
  5: #include <petscdmda.h>

  7: int main(int argc, char **argv)
  8: {
  9:   PetscInt       M = 6, N = 5, m = PETSC_DECIDE, n = PETSC_DECIDE, i, j, is, js, in, jen;
 10:   DM             da;
 11:   Vec            local, global;
 12:   PetscScalar ***l;

 14:   PetscFunctionBeginUser;
 15:   PetscCall(PetscInitialize(&argc, &argv, (char *)0, help));
 16:   /* Create distributed array and get vectors */
 17:   PetscCall(DMDACreate2d(PETSC_COMM_WORLD, DM_BOUNDARY_NONE, DM_BOUNDARY_NONE, DMDA_STENCIL_BOX, M, N, m, n, 3, 1, NULL, NULL, &da));
 18:   PetscCall(DMSetFromOptions(da));
 19:   PetscCall(DMSetUp(da));
 20:   PetscCall(DMCreateGlobalVector(da, &global));
 21:   PetscCall(DMCreateLocalVector(da, &local));

 23:   PetscCall(DMDAGetCorners(da, &is, &js, 0, &in, &jen, 0));
 24:   PetscCall(DMDAVecGetArrayDOF(da, local, &l));
 25:   for (i = is; i < is + in; i++) {
 26:     for (j = js; j < js + jen; j++) {
 27:       l[j][i][0] = 3 * (i + j * M);
 28:       l[j][i][1] = 3 * (i + j * M) + 1;
 29:       l[j][i][2] = 3 * (i + j * M) + 2;
 30:     }
 31:   }
 32:   PetscCall(DMDAVecRestoreArrayDOF(da, local, &l));
 33:   PetscCall(DMLocalToGlobalBegin(da, local, ADD_VALUES, global));
 34:   PetscCall(DMLocalToGlobalEnd(da, local, ADD_VALUES, global));

 36:   PetscCall(VecView(global, PETSC_VIEWER_STDOUT_WORLD));

 38:   /* Free memory */
 39:   PetscCall(VecDestroy(&local));
 40:   PetscCall(VecDestroy(&global));
 41:   PetscCall(DMDestroy(&da));
 42:   PetscCall(PetscFinalize());
 43:   return 0;
 44: }
