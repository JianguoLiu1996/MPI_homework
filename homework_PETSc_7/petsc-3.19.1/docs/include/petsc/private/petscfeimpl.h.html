251: static inline void CoordinatesRefToReal(PetscInt dimReal, PetscInt dimRef, const PetscReal xi0[], const PetscReal v0[], const PetscReal J[], const PetscReal xi[], PetscReal x[])
252: {
253:   PetscInt d, e;

255:   for (d = 0; d < dimReal; ++d) {
256:     x[d] = v0[d];
257:     for (e = 0; e < dimRef; ++e) x[d] += J[d * dimReal + e] * (xi[e] - xi0[e]);
258:   }
259: }

261: static inline void CoordinatesRealToRef(PetscInt dimReal, PetscInt dimRef, const PetscReal xi0[], const PetscReal v0[], const PetscReal invJ[], const PetscReal x[], PetscReal xi[])
262: {
263:   PetscInt d, e;

265:   for (d = 0; d < dimRef; ++d) {
266:     xi[d] = xi0[d];
267:     for (e = 0; e < dimReal; ++e) xi[d] += invJ[d * dimReal + e] * (x[e] - v0[e]);
268:   }
269: }

271: static inline PetscErrorCode PetscFEInterpolate_Static(PetscFE fe, const PetscScalar x[], PetscFEGeom *fegeom, PetscInt q, PetscScalar interpolant[])
272: {
273:   PetscTabulation T;

275:   PetscFunctionBeginHot;
276:   PetscCall(PetscFEGetCellTabulation(fe, 0, &T));
277:   {
278:     const PetscReal *basis = T->T[0];
279:     const PetscInt   Nb    = T->Nb;
280:     const PetscInt   Nc    = T->Nc;
281:     for (PetscInt fc = 0; fc < Nc; ++fc) {
282:       interpolant[fc] = 0.0;
283:       for (PetscInt f = 0; f < Nb; ++f) interpolant[fc] += x[f] * basis[(q * Nb + f) * Nc + fc];
284:     }
285:   }
286:   PetscCall(PetscFEPushforward(fe, fegeom, 1, interpolant));
287:   PetscFunctionReturn(PETSC_SUCCESS);
288: }

308: static inline PetscErrorCode PetscFEInterpolateGradient_Static(PetscFE fe, PetscInt k, const PetscScalar x[], PetscFEGeom *fegeom, PetscInt q, PetscScalar interpolant[])
309: {
310:   PetscTabulation T;
311:   PetscInt        fc, f, d;

313:   PetscFunctionBeginHot;
314:   PetscCall(PetscFEGetCellTabulation(fe, k, &T));
315:   {
316:     const PetscReal *basisDer = T->T[1];
317:     const PetscReal *basisHes = k > 1 ? T->T[2] : NULL;
318:     const PetscInt   Nb       = T->Nb;
319:     const PetscInt   Nc       = T->Nc;
320:     const PetscInt   cdim     = T->cdim;

322:     PetscCheck(cdim == fegeom->dimEmbed, PETSC_COMM_SELF, PETSC_ERR_ARG_INCOMP, "Geometry dim %" PetscInt_FMT " must match tabulation dim %" PetscInt_FMT, fegeom->dimEmbed, cdim);
323:     for (fc = 0; fc < Nc; ++fc) {
324:       for (d = 0; d < cdim; ++d) interpolant[fc * cdim + d] = 0.0;
325:       for (f = 0; f < Nb; ++f) {
326:         for (d = 0; d < cdim; ++d) interpolant[fc * cdim + d] += x[f] * basisDer[((q * Nb + f) * Nc + fc) * cdim + d];
327:       }
328:     }
329:     if (k > 1) {
330:       const PetscInt off = Nc * cdim;

332:       for (fc = 0; fc < Nc; ++fc) {
333:         for (d = 0; d < cdim * cdim; ++d) interpolant[off + fc * cdim * cdim + d] = 0.0;
334:         for (f = 0; f < Nb; ++f) {
335:           for (d = 0; d < cdim * cdim; ++d) interpolant[off + fc * cdim + d] += x[f] * basisHes[((q * Nb + f) * Nc + fc) * cdim * cdim + d];
336:         }
337:       }
338:     }
339:   }
340:   PetscCall(PetscFEPushforwardGradient(fe, fegeom, 1, interpolant));
341:   PetscFunctionReturn(PETSC_SUCCESS);
342: }

344: static inline PetscErrorCode PetscFEFreeInterpolateGradient_Static(PetscFE fe, const PetscReal basisDer[], const PetscScalar x[], PetscInt dim, const PetscReal invJ[], const PetscReal n[], PetscInt q, PetscScalar interpolant[])
345: {
346:   PetscReal   realSpaceDer[3];
347:   PetscScalar compGradient[3];
348:   PetscInt    Nb, Nc, fc, f, d, g;

350:   PetscFunctionBeginHot;
351:   PetscCall(PetscFEGetDimension(fe, &Nb));
352:   PetscCall(PetscFEGetNumComponents(fe, &Nc));

354:   for (fc = 0; fc < Nc; ++fc) {
355:     interpolant[fc] = 0.0;
356:     for (d = 0; d < dim; ++d) compGradient[d] = 0.0;
357:     for (f = 0; f < Nb; ++f) {
358:       for (d = 0; d < dim; ++d) {
359:         realSpaceDer[d] = 0.0;
360:         for (g = 0; g < dim; ++g) realSpaceDer[d] += invJ[g * dim + d] * basisDer[((q * Nb + f) * Nc + fc) * dim + g];
361:         compGradient[d] += x[f] * realSpaceDer[d];
362:       }
363:     }
364:     if (n) {
365:       for (d = 0; d < dim; ++d) interpolant[fc] += compGradient[d] * n[d];
366:     } else {
367:       for (d = 0; d < dim; ++d) interpolant[d] = compGradient[d];
368:     }
369:   }
370:   PetscFunctionReturn(PETSC_SUCCESS);
371: }

373: static inline PetscErrorCode PetscFEInterpolateFieldAndGradient_Static(PetscFE fe, PetscInt k, const PetscScalar x[], PetscFEGeom *fegeom, PetscInt q, PetscScalar interpolant[], PetscScalar interpolantGrad[])
374: {
375:   PetscTabulation T;
376:   PetscInt        fc, f, d;

378:   PetscFunctionBeginHot;
379:   PetscCall(PetscFEGetCellTabulation(fe, k, &T));
380:   {
381:     const PetscReal *basis    = T->T[0];
382:     const PetscReal *basisDer = T->T[1];
383:     const PetscReal *basisHes = k > 1 ? T->T[2] : NULL;
384:     const PetscInt   Nb       = T->Nb;
385:     const PetscInt   Nc       = T->Nc;
386:     const PetscInt   cdim     = T->cdim;

388:     PetscCheck(cdim == fegeom->dimEmbed, PETSC_COMM_SELF, PETSC_ERR_ARG_INCOMP, "Geometry dim %" PetscInt_FMT " must match tabulation dim %" PetscInt_FMT, fegeom->dimEmbed, cdim);
389:     for (fc = 0; fc < Nc; ++fc) {
390:       interpolant[fc] = 0.0;
391:       for (d = 0; d < cdim; ++d) interpolantGrad[fc * cdim + d] = 0.0;
392:       for (f = 0; f < Nb; ++f) {
393:         interpolant[fc] += x[f] * basis[(q * Nb + f) * Nc + fc];
394:         for (d = 0; d < cdim; ++d) interpolantGrad[fc * cdim + d] += x[f] * basisDer[((q * Nb + f) * Nc + fc) * cdim + d];
395:       }
396:     }
397:     if (k > 1) {
398:       const PetscInt off = Nc * cdim;

400:       for (fc = 0; fc < Nc; ++fc) {
401:         for (d = 0; d < cdim * cdim; ++d) interpolantGrad[off + fc * cdim * cdim + d] = 0.0;
402:         for (f = 0; f < Nb; ++f) {
403:           for (d = 0; d < cdim * cdim; ++d) interpolantGrad[off + fc * cdim + d] += x[f] * basisHes[((q * Nb + f) * Nc + fc) * cdim * cdim + d];
404:         }
405:       }
406:       PetscCall(PetscFEPushforwardHessian(fe, fegeom, 1, &interpolantGrad[off]));
407:     }
408:   }
409:   PetscCall(PetscFEPushforward(fe, fegeom, 1, interpolant));
410:   PetscCall(PetscFEPushforwardGradient(fe, fegeom, 1, interpolantGrad));
411:   PetscFunctionReturn(PETSC_SUCCESS);
412: }
