123:                    _aa->grid->mycol == _bb->grid->mycol, \
124:                  PetscObjectComm((PetscObject)(a)), PETSC_ERR_ARG_INCOMP, "Arguments #%d and #%d have different ScaLAPACK distribution", arga, argb); \
