239: static inline PetscErrorCode SNESLogConvergenceHistory(SNES snes, PetscReal res, PetscInt its)
240: {
241:   PetscFunctionBegin;
242:   PetscCall(PetscObjectSAWsTakeAccess((PetscObject)snes));
243:   if (snes->conv_hist && snes->conv_hist_max > snes->conv_hist_len) {
244:     if (snes->conv_hist) snes->conv_hist[snes->conv_hist_len] = res;
245:     if (snes->conv_hist_its) snes->conv_hist_its[snes->conv_hist_len] = its;
246:     snes->conv_hist_len++;
247:   }
248:   PetscCall(PetscObjectSAWsGrantAccess((PetscObject)snes));
249:   PetscFunctionReturn(PETSC_SUCCESS);
250: }

287: /*
288:     Either generate an error or mark as diverged when a real from a SNES function norm is Nan or Inf.
289:     domainerror is reset here, once reason is set, to allow subsequent iterations to be feasible (e.g. line search).
290: */
291: #define SNESCheckFunctionNorm(snes, beta) \
292:   do { \
293:     if (PetscIsInfOrNanReal(beta)) { \
294:       PetscCheck(!snes->errorifnotconverged, PetscObjectComm((PetscObject)snes), PETSC_ERR_NOT_CONVERGED, "SNESSolve has not converged due to Nan or Inf norm"); \
295:       { \
296:         PetscBool domainerror; \
297:         PetscCall(MPIU_Allreduce(&snes->domainerror, &domainerror, 1, MPIU_BOOL, MPI_LOR, PetscObjectComm((PetscObject)snes))); \
298:         if (domainerror) { \
299:           snes->reason      = SNES_DIVERGED_FUNCTION_DOMAIN; \
300:           snes->domainerror = PETSC_FALSE; \
301:         } else snes->reason = SNES_DIVERGED_FNORM_NAN; \
302:         PetscFunctionReturn(PETSC_SUCCESS); \
303:       } \
304:     } \
305:   } while (0)

307: #define SNESCheckJacobianDomainerror(snes) \
308:   do { \
309:     if (snes->checkjacdomainerror) { \
310:       PetscBool domainerror; \
311:       PetscCall(MPIU_Allreduce(&snes->jacobiandomainerror, &domainerror, 1, MPIU_BOOL, MPI_LOR, PetscObjectComm((PetscObject)snes))); \
312:       if (domainerror) { \
313:         snes->reason = SNES_DIVERGED_JACOBIAN_DOMAIN; \
314:         PetscCheck(!snes->errorifnotconverged, PetscObjectComm((PetscObject)snes), PETSC_ERR_NOT_CONVERGED, "SNESSolve has not converged due to Jacobian domain error"); \
315:         PetscFunctionReturn(PETSC_SUCCESS); \
316:       } \
317:     } \
318:   } while (0)

320: #define SNESCheckKSPSolve(snes) \
321:   do { \
322:     KSPConvergedReason kspreason; \
323:     PetscInt           lits; \
324:     PetscCall(KSPGetIterationNumber(snes->ksp, &lits)); \
325:     snes->linear_its += lits; \
326:     PetscCall(KSPGetConvergedReason(snes->ksp, &kspreason)); \
327:     if (kspreason < 0) { \
328:       if (kspreason == KSP_DIVERGED_NANORINF) { \
329:         PetscBool domainerror; \
330:         PetscCall(MPIU_Allreduce(&snes->domainerror, &domainerror, 1, MPIU_BOOL, MPI_LOR, PetscObjectComm((PetscObject)snes))); \
331:         if (domainerror) snes->reason = SNES_DIVERGED_FUNCTION_DOMAIN; \
332:         else snes->reason = SNES_DIVERGED_LINEAR_SOLVE; \
333:         PetscFunctionReturn(PETSC_SUCCESS); \
334:       } else { \
335:         if (++snes->numLinearSolveFailures >= snes->maxLinearSolveFailures) { \
336:           PetscCall(PetscInfo(snes, "iter=%" PetscInt_FMT ", number linear solve failures %" PetscInt_FMT " greater than current SNES allowed %" PetscInt_FMT ", stopping solve\n", snes->iter, snes->numLinearSolveFailures, snes->maxLinearSolveFailures)); \
337:           snes->reason = SNES_DIVERGED_LINEAR_SOLVE; \
338:           PetscFunctionReturn(PETSC_SUCCESS); \
339:         } \
340:       } \
341:     } \
342:   } while (0)
