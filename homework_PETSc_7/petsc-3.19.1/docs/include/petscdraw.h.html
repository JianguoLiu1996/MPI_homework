 72: /*MC

 74:    PetscDrawRealToColor - Maps a real value within an interval to a color.
 75:    The color is an integer value in the range [`PETSC_DRAW_BASIC_COLORS` to 255]
 76:    that can be passed to various drawing routines.

 78:    Synopsis:
 79: #include <petscdraw.h>
 80:    int PetscDrawRealToColor(PetscReal value,PetscReal min,PetscReal max)

 82:    Not Collective

 84:    Input Parameters:
 85: +  value - value to map within the interval [`min`, `max`]
 86: .  min - lower end of interval
 87: -  max - upper end of interval

 89:    Returns:
 90:    The result as integer

 92:    Level: intermediate

 94:    Note:
 95:    Values outside the interval [`min`, `max`] are clipped.

 97: .seealso: `PetscDraw`, `PetscDrawPointPixel()`, `PetscDrawPoint()`, `PetscDrawLine()`, `PetscDrawTriangle()`, `PetscDrawRectangle()`
 98: M*/
 99: static inline int PetscDrawRealToColor(PetscReal value, PetscReal min, PetscReal max)
100: {
101:   value = PetscClipInterval(value, min, max);
102:   return PETSC_DRAW_BASIC_COLORS + (int)((255 - PETSC_DRAW_BASIC_COLORS) * (value - min) / (max - min));
103: }

354: PETSC_EXTERN jmp_buf PetscXIOErrorHandlerJumpBuf;

359:   /*MC
360:     PetscDrawCollectiveBegin - Begins a set of draw operations

362:    Collective

364:    Synopsis:
365: #include <petscdraw.h>
366:     PetscErrorCode PetscDrawCollectiveBegin(PetscDraw draw)

368:     Collective

370:    Input Parameter:
371: .   draw - the draw object

373:   Level: advanced

375:    Notes:
376:     This is a macro that handles its own error checking, it does not return an error code.

378:     The set of operations needs to be ended by a call to `PetscDrawCollectiveEnd()`.

380:     X windows draw operations that are enclosed by these routines handle correctly resizing or closing of
381:     the window without crashing the program.

383:   Developer Note:
384:     This only applies to X windows and so should have a more specific name such as `PetscDrawXCollectiveBegin()`

386: .seealso: `PetscDrawCollectiveEnd()`
387: M*/
388:   #define PetscDrawCollectiveBegin(draw) \
389:     do { \
390:       jmp_buf                       _Petsc_jmpbuf; \
391:       volatile PetscXIOErrorHandler _Petsc_xioerrhdl = PETSC_NULLPTR; \
392:       PetscBool                     _Petsc_isdrawx, _Petsc_xioerr, _Petsc_xioerr_local = PETSC_FALSE; \
393:       PetscCall(PetscObjectTypeCompare((PetscObject)(draw), PETSC_DRAW_X, &_Petsc_isdrawx)); \
394:       if (_Petsc_isdrawx) { \
395:         PetscCall(PetscMemcpy(&_Petsc_jmpbuf, &PetscXIOErrorHandlerJumpBuf, sizeof(_Petsc_jmpbuf))); \
396:         _Petsc_xioerrhdl = PetscSetXIOErrorHandler(PetscXIOErrorHandlerJump); \
397:         if (setjmp(PetscXIOErrorHandlerJumpBuf)) { \
398:           _Petsc_xioerr_local = PETSC_TRUE; \
399:           do { \
400:             PetscDrawCollectiveEnd(draw); \
401:           } \
402:         } \
403:         do { \
404:       } while (0)

406:   /*MC
407:     PetscDrawCollectiveEnd - Ends a set of draw operations begun with `PetscDrawCollectiveBegin()`

409:    Collective

411:    Synopsis:
412: #include <petscdraw.h>
413:     PetscErrorCode PetscDrawCollectiveEnd(PetscDraw draw)

415:     Collective

417:   Input Parameter:
418: .   draw - the draw object

420:   Level: advanced

422:   Notes:
423:     This is a macro that handles its own error checking, it does not return an error code.

425:     X windows draw operations that are enclosed by these routines handle correctly resizing or closing of
426:     the window without crashing the program.

428:   Developer Note:
429:     This only applies to X windows and so should have a more specific name such as `PetscDrawXCollectiveEnd()`

431: .seealso: `PetscDrawCollectiveBegin()`
432: M*/
433:   #define PetscDrawCollectiveEnd(draw) \
434:     if (_Petsc_isdrawx) { \
435:       (void)PetscSetXIOErrorHandler(_Petsc_xioerrhdl); \
436:       PetscCall(PetscMemcpy(&PetscXIOErrorHandlerJumpBuf, &_Petsc_jmpbuf, sizeof(PetscXIOErrorHandlerJumpBuf))); \
437:       PetscCall(MPIU_Allreduce(&_Petsc_xioerr_local, &_Petsc_xioerr, 1, MPIU_BOOL, MPI_LOR, PetscObjectComm((PetscObject)(draw)))); \
