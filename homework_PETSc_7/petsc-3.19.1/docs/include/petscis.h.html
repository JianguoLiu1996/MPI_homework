287: /*@C
288:      PetscLayoutFindOwner - Find the owning rank for a global index

290:     Not Collective; No Fortran Support

292:    Input Parameters:
293: +    map - the layout
294: -    idx - global index to find the owner of

296:    Output Parameter:
297: .    owner - the owning rank

299:    Level: developer

301: .seealso: `PetscLayout`, `PetscLayoutFindOwnerIndex()`
302: @*/
303: static inline PetscErrorCode PetscLayoutFindOwner(PetscLayout map, PetscInt idx, PetscMPIInt *owner)
304: {
305:   PetscMPIInt lo = 0, hi, t;

307:   PetscFunctionBegin;
308:   *owner = -1; /* GCC erroneously issues warning about possibly uninitialized use when error condition */
309: #if defined(PETSC_USE_DEBUG)
310:   PetscCheck((map->n >= 0) && (map->N >= 0) && (map->range), PETSC_COMM_SELF, PETSC_ERR_ARG_WRONGSTATE, "PetscLayoutSetUp() must be called first");
311:   PetscCheck(idx >= 0 && idx <= map->N, PETSC_COMM_SELF, PETSC_ERR_ARG_OUTOFRANGE, "Index %" PetscInt_FMT " is out of range", idx);
312: #endif
313:   hi = map->size;
314:   while (hi - lo > 1) {
315:     t = lo + (hi - lo) / 2;
316:     if (idx < map->range[t]) hi = t;
317:     else lo = t;
318:   }
319:   *owner = lo;
320:   PetscFunctionReturn(PETSC_SUCCESS);
321: }

323: /*@C
324:      PetscLayoutFindOwnerIndex - Find the owning MPI rank and the local index on that rank for a global index

326:     Not Collective; No Fortran Support

328:    Input Parameters:
329: +    map   - the layout
330: -    idx   - global index to find the owner of

332:    Output Parameters:
333: +    owner - the owning rank
334: -    lidx  - local index used by the owner for `idx`

336:    Level: developer

338: .seealso: `PetscLayout`, `PetscLayoutFindOwner()`
339: @*/
340: static inline PetscErrorCode PetscLayoutFindOwnerIndex(PetscLayout map, PetscInt idx, PetscMPIInt *owner, PetscInt *lidx)
341: {
342:   PetscMPIInt lo = 0, hi, t;

344:   PetscFunctionBegin;
345: #if defined(PETSC_USE_DEBUG)
346:   PetscCheck((map->n >= 0) && (map->N >= 0) && (map->range), PETSC_COMM_SELF, PETSC_ERR_ARG_WRONGSTATE, "PetscLayoutSetUp() must be called first");
347:   PetscCheck(idx >= 0 && idx <= map->N, PETSC_COMM_SELF, PETSC_ERR_ARG_OUTOFRANGE, "Index %" PetscInt_FMT " is out of range", idx);
348: #endif
349:   hi = map->size;
350:   while (hi - lo > 1) {
351:     t = lo + (hi - lo) / 2;
352:     if (idx < map->range[t]) hi = t;
353:     else lo = t;
354:   }
355:   if (owner) *owner = lo;
356:   if (lidx) *lidx = idx - map->range[lo];
357:   PetscFunctionReturn(PETSC_SUCCESS);
358: }
